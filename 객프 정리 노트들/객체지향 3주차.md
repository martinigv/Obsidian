## 16Slide
1. class 이름은 Rectangle
2. width, height int타입의 맴버 변수 필요
3. getArea() 맴버 함수 필요
```c++
#include <iostream>
using namespace std;
class Rectangle {
public:
	int width;
	int height;

	int getArea() {
		return width * height;
	}
};


int main() {
	Rectangle rect;
	rect.width = 3;
	rect.height = 5;
	cout << "사각형의 면적은 " << rect.getArea() << endl;
}
```

## 30slide(3-6 실습)
1. width, height 변수 필요(가로, 세로)
2. 생성자 2개( Rectangle(n), Rectangle(w,h))(**Rectangle()은 Rectangle(n)에 위임**)
3. isSquare() 맴버 함수 필요(bool 타입)
4. isSquare() 구현
```c++
#include <iostream>
using namespace std;

class Rectangle {
public:
	int width;
	int height;

	Rectangle(int w, int h) {
		width = w;
		height = h;
	}
	Rectangle() : Rectangle(1) {}
	Rectangle(int n) {
		width = n;
		height = n;
	}
	bool isSquare() {
		return width == height;
	}
};

int main() {
	Rectangle rect1;
	Rectangle rect2(3, 5);
	Rectangle rect3(3);
	if (rect1.isSquare()) cout << "rect1은 정사각형이다." << endl;
	if (rect2.isSquare()) cout << "rect2는 정사각형이다." << endl;
	if (rect3.isSquare()) cout << "rect3는 정사각형이다." << endl;
}

```
